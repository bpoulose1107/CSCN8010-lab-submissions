import numpy as np
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score, mean_absolute_error

# To define a function to calculate MAPE
def mean_absolute_percentage_error(y_true, y_pred): 
    y_true, y_pred = np.array(y_true), np.array(y_pred)
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100 if np.all(y_true != 0) else np.nan

# Extracting BMI and target from datasets
X_train_bmi = train_set[['bmi']].values
y_train = train_set['target'].values
X_val_bmi = validation_set[['bmi']].values
y_val = validation_set['target'].values

# Lists to store results for each polynomial degree
degrees = range(6)
results = []

for degree in degrees:
    # Create polynomial features
    poly = PolynomialFeatures(degree=degree)
    X_train_poly = poly.fit_transform(X_train_bmi)
    X_val_poly = poly.transform(X_val_bmi)
    
    # Fit a linear regression model
    model = LinearRegression()
    model.fit(X_train_poly, y_train)
    
    # Predicting on training and validation data
    y_train_pred = model.predict(X_train_poly)
    y_val_pred = model.predict(X_val_poly)
    
    # Calculate performance metrics
    train_r2 = r2_score(y_train, y_train_pred)
    val_r2 = r2_score(y_val, y_val_pred)
    train_mae = mean_absolute_error(y_train, y_train_pred)
    val_mae = mean_absolute_error(y_val, y_val_pred)
    train_mape = mean_absolute_percentage_error(y_train, y_train_pred)
    val_mape = mean_absolute_percentage_error(y_val, y_val_pred)
    
    # Store results in a dictionary
    results.append({
        'Degree': degree,
        'Train R^2': train_r2,
        'Validation R^2': val_r2,
        'Train MAE': train_mae,
        'Validation MAE': val_mae,
        'Train MAPE': train_mape,
        'Validation MAPE': val_mape
    })

# To display results
print("Degree | Train R^2 | Validation R^2 | Train MAE | Validation MAE | Train MAPE | Validation MAPE")
for result in results:
    print(f"{result['Degree']}      | {result['Train R^2']:.4f}   | {result['Validation R^2']:.4f}         | {result['Train MAE']:.2f}      | {result['Validation MAE']:.2f}         | {result['Train MAPE']:.2f}%      | {result['Validation MAPE']:.2f}%")
